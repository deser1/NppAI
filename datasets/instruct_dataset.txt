def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n-1)

[USER]: Lista todo w Vue
[AI]:
<!DOCTYPE html>
<html lang="pl">
<head>
    <meta charset="UTF-8">
    <title>Vue To-Do List</title>
    <script src="https://unpkg.com/vue@3/dist/vue.global.js"></script>
    <script src="https://cdn.tailwindcss.com"></script>
</head>
<body class="bg-gray-100 p-10 font-sans">
    <div id="app" class="max-w-md mx-auto bg-white rounded-xl shadow-md overflow-hidden p-6">
        <h2 class="text-2xl font-bold mb-4 text-gray-800">Moje Zadania</h2>
        <div class="flex mb-4">
            <input v-model="newTask" @keyup.enter="addTask" placeholder="Dodaj nowe zadanie..." class="flex-1 border rounded-l-lg p-2 outline-none focus:ring-2 focus:ring-blue-400">
            <button @click="addTask" class="bg-blue-500 text-white px-4 rounded-r-lg hover:bg-blue-600 transition">Dodaj</button>
        </div>
        <ul>
            <li v-for="(task, index) in tasks" :key="index" class="flex items-center justify-between p-3 border-b last:border-0 hover:bg-gray-50 transition">
                <span :class="{'line-through text-gray-400': task.done}" class="cursor-pointer flex-1" @click="task.done = !task.done">{{ task.text }}</span>
                <button @click="removeTask(index)" class="text-red-500 hover:text-red-700 font-bold ml-4">X</button>
            </li>
        </ul>
        <p v-if="tasks.length === 0" class="text-gray-500 text-center mt-4">Brak zadań. Dodaj coś!</p>
    </div>

    <script>
        const { createApp, ref } = Vue;
        createApp({
            setup() {
                const newTask = ref('');
                const tasks = ref([
                    { text: 'Nauczyć się Vue 3', done: false },
                    { text: 'Stworzyć model AI', done: true }
                ]);
                const addTask = () => {
                    if (newTask.value.trim()) {
                        tasks.value.push({ text: newTask.value, done: false });
                        newTask.value = '';
                    }
                };
                const removeTask = (index) => {
                    tasks.value.splice(index, 1);
                };
                return { newTask, tasks, addTask, removeTask }
            }
        }).mount('#app');
    </script>
</body>
</html>

<!DOCTYPE html>
<html lang="pl">
<head>
    <meta charset="UTF-8">
    <title>Licznik Vue</title>
    <script src="https://unpkg.com/vue@3/dist/vue.global.js"></script>
    <script src="https://cdn.tailwindcss.com"></script>
</head>
<body class="bg-gray-200 h-screen flex items-center justify-center font-sans">
    <div id="app" class="bg-white rounded-3xl shadow-xl p-10 text-center w-80">
        <h2 class="text-2xl font-bold text-gray-700 mb-6">Licznik (Counter)</h2>
        <div class="text-6xl font-black text-blue-600 mb-8">{{ count }}</div>
        <div class="flex justify-center gap-4">
            <button @click="decrement" class="bg-red-500 hover:bg-red-600 text-white w-12 h-12 rounded-full text-2xl font-bold transition shadow-lg">-</button>
            <button @click="reset" class="bg-gray-500 hover:bg-gray-600 text-white px-4 rounded-lg font-semibold transition shadow-lg">Reset</button>
            <button @click="increment" class="bg-green-500 hover:bg-green-600 text-white w-12 h-12 rounded-full text-2xl font-bold transition shadow-lg">+</button>
        </div>
    </div>

    <script>
        const { createApp, ref } = Vue;
        createApp({
            setup() {
                const count = ref(0);
                const increment = () => count.value++;
                const decrement = () => count.value--;
                const reset = () => count.value = 0;
                return { count, increment, decrement, reset }
            }
        }).mount('#app');
    </script>
</body>
</html>
